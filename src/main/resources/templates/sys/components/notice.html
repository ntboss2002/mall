notice 通知公告组件

			// Step1. 引入依赖
			let notice = layui.notice;.

			

			// Step2. 使用
			$("#notice").click(function () {
				notice.success("消息提示");
			});

			$("#notice2").click(function () {
				notice.info("消息提示");
			});

			$("#notice3").click(function () {
				notice.warning("消息提示");
			});

			$("#notice4").click(function () {
				notice.error("消息提示");
			});
